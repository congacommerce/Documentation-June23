        </apt-product-type-filter>

        <apt-category-filter [category]="category" title="{{'PRODUCT_LIST.RELATED_CATEGORIES' | translate}}" limit="10"
          relationship="peers" class="mb-2" (onChange)="onCategory($event)"></apt-category-filter>

        <apt-category-filter [category]="category" title="{{'COMMON.SUB_CATEGORIES' | translate}}" limit="10"

        </apt-category-filter>

        <div class="card mb-2">
          <div class="card-body">
            <h5 class="card-title">{{'PRODUCT_LIST.PRODUCT_FAMILY' | translate}}</h5>

              (onChange)="handlePicklistChange($event)"></apt-input-select>
          </div>
        </div>
      </div>
    </div>
    <div class="col-12 col-lg-9">
      <div *ngIf="!hasSearchError else showSearchError">

        <div *ngIf="data$ | async as data;else loading">
          <nav class="navbar navbar-light bg-white p-0 mb-3">
            <pl-results [recordCount]="data?.TotalCount" [limit]="pageSize" [offset]="(page - 1) * pageSize"
              [page]="page" [view]="view" [query]="searchString" [sortBy]="sortField" (onViewChange)="view = $event"
              (onSortChange)="onSortChange($event)" (onPageSizeChange)="onPageSizeChange($event)" class="d-block w-100">
            </pl-results>
          </nav>

          <div class="d-flex justify-content-center" *ngIf="data?.TotalCount">
            <pagination (pageChanged)="onPage($event)" [totalItems]="data?.TotalCount" [(ngModel)]="page"
              [boundaryLinks]="true" [itemsPerPage]="pageSize" [maxSize]="5" [firstText]="paginationButtonLabels.first"
              [previousText]="paginationButtonLabels.previous" [nextText]="paginationButtonLabels.next"
              [lastText]="paginationButtonLabels.last"></pagination>
          </div>

          <div class="row" *ngIf="view == 'grid' && data?.Products">
            <div class="col-12 col-sm-6 col-md-4 col-xl-3 mb-3" *ngFor="let product of data?.Products">
              <apt-product-card [product]="product"></apt-product-card>
            </div>
          </div>
          <div class="row" *ngIf="view == 'list' && data?.Products">
            <div class="col-12">
              <ul class="list-unstyled list-group">

                </li>
              </ul>
            </div>
          </div>

          <div class="d-flex justify-content-center" *ngIf="data?.TotalCount">
            <pagination (pageChanged)="onPage($event)" [totalItems]="data?.TotalCount" [(ngModel)]="page"
              [boundaryLinks]="true" [itemsPerPage]="pageSize" [maxSize]="5" [firstText]="paginationButtonLabels.first"
              [previousText]="paginationButtonLabels.previous" [nextText]="paginationButtonLabels.next"

          </div>
        </div>
      </div>
    </div>
