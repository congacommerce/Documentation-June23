    /**
     * Method clears data from cache and database. When removeReference is set to true, it will also remove the cache reference from memory
     * and also break any existing observable streams to the interface.
     * @param keyList The list of keys to clear
     * @param removeReference Set to true to remove cache pointers from memory
     */
    clearCache(keyList: Array<string>, removeReference: boolean = false): Promise<void> {
        if (removeReference)
            this.cache = _.filter(this.cache, c => !_.includes(keyList, _.get(c, 'id')));
        _.forEach(_.filter(this.cache, c => _.includes(keyList, _.get(c, 'id'))), cacheRecord => {
            cacheRecord.subject.next(undefined);
            cacheRecord.data = null;
        });
        return this.clearDatabase(keyList);
    }

    /**
     * Method will clear records from the indexeddb. If no keylist is specified it will clear the whole database.
     * @param keyList The keys to clear
     */
    clearDatabase(keyList?: Array<string>): Promise<any> {
        if (this.db) {
            if (!_.isEmpty(keyList)) {
                return Promise.all(_.map(keyList, key => this.db.delete(this.DB_STORE, key)));
            } else {
                return this.db.clear(this.DB_STORE);
            }
        } else {
            return Promise.resolve(true);
        }
    }

    /*
     * Method clears the whole cache
     */
    clear(): Observable<void> {
        this.cache = [];
        return from(this.clearDatabase());
    }


    _set(key: string, data: any, local: boolean = false) {
        this._cache[key] = data;

    _get(key: string) {
        if (this._cache[key])
            return this._cache[key];
        else if (localStorage.getItem(key))
            try {
                return JSON.parse(localStorage.getItem(key));
            } catch (e) {
                return localStorage.getItem(key);
            }
        else
            return null;
    }

    get(type: ClassType<AObject>, key: string, executable: () => Observable<any>, useIndexedDB: boolean = false, skipCache: boolean = false, parse: boolean = true, strategy: 'freshness' | 'performance' = 'freshness'): Observable<any> {
        const hashKey = SalesforceUtils.hashCode(key);
        let cacheRecord = _.find(this.cache, (c) => c.id === hashKey && c.type === type);
        if (_.isNil(cacheRecord)) {
            const subject = new BehaviorSubject<Array<AObject> | AObject>(undefined);
            cacheRecord = {
                id: hashKey,
                type: type,
                executable: executable,
                subject: subject,
                output: subject.pipe(
                    filter(r => !_.isUndefined(r))
                    , distinctUntilChanged((a, b) => this.isEqual(a, b))
                    , map(r => _.cloneDeep(r))
                    , tap(data => {
                        if (cacheRecord.skipCache) {
                            _.remove(this.cache, c => c.id === cacheRecord.id);
                        } else {
                            cacheRecord.data = data;
                        }
                    })
                ),
                parse: parse,
                skipCache: skipCache,
                strategy: strategy
            };
            this.cache.push(cacheRecord);
            this.publish(cacheRecord, useIndexedDB);
        }
        return cacheRecord.output;
    }


    buffer(bufferKey: string, value: any, executable: (arg) => Observable<any>, dataMap?: (arg) => any, bufferTimeInMilliseconds?: number, bufferCount?: number, debounceResult: boolean = false): Observable<any> {
        let bufferData = _.find(_.get(this.bufferCache, bufferKey), d => _.get(d, 'state.value') === 'open');

        bufferTimeInMilliseconds = _.defaultTo(bufferTimeInMilliseconds, this.configurationService.get('disableBuffer', false) === true ? 0 : this.configurationService.get('bufferTime', 50));
        bufferCount = _.defaultTo(bufferCount, this.configurationService.get('disableBuffer', false) === true ? 1 : this.configurationService.get('maxBufferSize', 10));


        if (_.isNil(bufferData)) {
            const state = new BehaviorSubject<'open' | 'running' | 'closed'>('open');
            let acc = new BehaviorSubject<any>(value).pipe(
                bufferTime(bufferTimeInMilliseconds, undefined, bufferCount)
            );
            if (debounceResult) {
                acc = new BehaviorSubject<any>(value).pipe(
                    tap(d => _.set(bufferData, 'data', _.concat(_.get(bufferData, 'data', []), d))),
                    debounceTime(this.configurationService.get('debounceTime', 1000)),
                    map(() => _.get(bufferData, 'data'))
                );
            }
            bufferData = {
                state: state,
                accumulator: acc
                    .pipe(
                        filter(x => !_.isEmpty(x)),
                        takeUntil(state.pipe(filter(s => s === 'closed'))),
                        mergeMap(data => {
                            state.next('running');
                            return bufferData.executable(_.flatten(data));
                        }),
                        tap(d => {
                            state.next('closed');
                            _.remove(_.get(this.bufferCache, bufferKey), x => !_.includes(['open', 'running'], _.get(x, 'state.value')));
                        }),
                        share()),
                executable: executable
            };
            _.set(this.bufferCache, bufferKey, _.concat(_.get(this.bufferCache, bufferKey, []), bufferData));
        } else
            bufferData.accumulator.next(value);


        return bufferData.accumulator.pipe(
            map(data => {
                return (!_.isNil(dataMap)) ? dataMap(data) : data;
            })
        );
    }

    /**
     * @description
     * Method is similar to refresh, but more generic.
     * looks for all cache items related to the input and attempts to update the state.
     * Not perfect, may not update all data everywhere in the app (i.e. child data). Useful for top level data
     * If no data is passed into the state, it will pull the latest from the server.
     *
     * @param key a string, number or class type to update.
     * @param newState optional state data. If null, will update cache with server data. When not null, will update cache with client data only.
     * @param executable optional executable function. Will replace existing executable funciton in the cache.
     * @param relatedRecords when set to true, will update state for related records (default true)
     */
    update(key: string | number | ClassType<AObject>, newState?: AObject | Array<AObject>, relatedRecords: boolean = true) {
        let cacheRecords = this.getCacheRecords(key, relatedRecords, 'freshness');
        const keys = this.getKeysForRecords(cacheRecords);
        const fullKeyList = _.concat(keys, _.map(cacheRecords, 'id'));
        this.clearDatabase(fullKeyList).then(() => {
            _.forEach(_.compact(cacheRecords), c => {
                let oldData = _.cloneDeep(_.get(c, 'data'));

                // Immediatley publish the new state to cache records if one is provided
                if (!_.isNil(newState)) {
                    const handleRecord = (record: AObject) => {
                        _.set(record, 'LastModifiedDate', new Date());
                        // Iterate through the cache record data
                        if (_.isArray(oldData)) {
                            let hasId: boolean = false;
                            _.forEach(oldData, (od, index) => {
                                if (_.get(od, 'Id') === _.get(record, 'Id')) {
                                    oldData.splice(index, 1, record);
                                    hasId = true;
                                }
                            });
                            if (!hasId)
                                oldData.push(record);
                        } else if (_.isObject(oldData) && _.get(oldData, 'Id') === _.get(record, 'Id')) {
                            _.set(c, 'data', record);
                        }
                    };

                    if (_.isArray(newState)) {
                        _.forEach(newState, r => handleRecord(r));
                    } else if (_.isObject(newState))
                        handleRecord(newState as AObject);
                    this.push(c.id, oldData);
                } else {
                    c.dirty = true;
                    this.publish(c, true, false);
                }
            });
        });
    }

    /**
     * Method immediatley pushes a new value to existing cache subscribers
     * @param hashKey The cache key identifier to update
     * @param data The data to push to the cache subject
     */
    push(hashKey: number, data: any): void {
        const callback = () => {
            const cacheRecord = _.find(this.cache, c => c.id === hashKey);
            if (!_.isNil(cacheRecord)) {
                cacheRecord.subject.next(undefined);
                if (!_.isEmpty(cacheRecord.subject.observers))
                    this.ngZone.run(() => cacheRecord.subject.next(data));
                else
                    this.cache = _.filter(this.cache, c => c.id !== cacheRecord.id);
            }
        };
        if (this.db)
            this.db.delete(this.DB_STORE, hashKey).then(() => callback(), () => callback());
        else
            callback();
    }

    getCacheRecords(key: string | number | ClassType<AObject>, related: boolean = false, strategy?: 'freshness' | 'performance') {
        // Garbage collection
        const empty = _.filter(this.cache, c => _.get(c, 'subject.observers.length') === 0);
        this.clearDatabase(_.map(empty, 'id')).then(() => _.remove(this.cache, c => _.includes(_.map(empty, 'id'), _.get(c, 'id'))));


        let cacheRecords = [];
        if (_.isString(key)) {
            const hashKey = SalesforceUtils.hashCode(key as string);
            cacheRecords.push(_.find(this.cache, c => c.id === hashKey));
        } else if (_.isNumber(key)) {
            cacheRecords.push(_.find(this.cache, c => c.id === key));
        } else {
            cacheRecords = _.filter(this.cache, c => c.type === key);
        }

        let relatedRecords = [];
        // Get child keys from data
        if (related === true) {
            relatedRecords = _.filter(this.cache, c => _.includes(this.getKeysForRecords(cacheRecords), c.id));
            if (!_.isNil(strategy))
                relatedRecords = _.filter(relatedRecords, r => r.strategy === strategy);

            cacheRecords = _.concat(cacheRecords, relatedRecords);
        }

        return _.uniqBy(cacheRecords, 'id');
    }

    getKeysForRecords(cacheRecordList: Array<CacheKey>) {
        const keyList: Set<string> = new Set<string>();
        const getKeys = (record: AObject) => {
            keyList.add(_.get(record, '_metadata.cacheKey'));
            _.forEach(_.values(_.pickBy(record, _.isArray)), recordList => {
                _.forEach(recordList, cr => getKeys(cr));
            });
            _.forEach(_.values(_.pickBy(record, r => _.isPlainObject(r) && r instanceof AObject)), cr => {
                getKeys(cr);
            });
        };
        _.forEach(cacheRecordList, c => {
            if (_.isArray(_.get(c, 'data')))
                _.forEach(_.get(c, 'data'), data => getKeys(data));
            else if (_.isPlainObject(_.get(c, 'data')) && _.get(c, 'data') instanceof AObject) {
                getKeys(_.get(c, 'data'));
            }
        });
        return _.compact(Array.from(keyList));
    }

    isEqual(oldData, newData): boolean {
        const undefinedAsNull = (k, v) => (v === undefined) ? null : v;
        const handleObject = (obj1, obj2) => {
            const arrayKeys = _.keys(_.pickBy(obj2, _.isArray));
            const objectKeys = _.keys(_.pickBy(obj2, o => o instanceof AObject));

            const arraysEqual = _.every(arrayKeys, key => handleArrays(_.get(obj1, key), _.get(obj2, key)));
            const objectsEqual = _.every(objectKeys, key => handleObject(_.get(obj1, key), _.get(obj2, key)));
            const dataEqual = (_.has(obj2, 'Id'))
                // tslint:disable-next-line:triple-equals
                ? (_.get(obj1, 'Id') == _.get(obj2, 'Id') && _.get(obj1, 'LastModifiedDate') == _.get(obj2, 'LastModifiedDate'))
                : _.isMatch(obj1, obj2);

            return (dataEqual && arraysEqual && objectsEqual);
        };

        const handleArrays = (arr1, arr2) => {
            return _.get(arr1, 'length') === _.get(arr2, 'length') && _.every(arr2, i => handleObject(_.find(arr1, elem => _.get(elem, 'Id') === _.get(i, 'Id')), i));
        };

        let lite = false;
        if (_.isNil(oldData) && !_.isNil(newData))
            return false;
        else {
            try {
                lite = JSON.stringify(oldData, undefinedAsNull) === JSON.stringify(newData, undefinedAsNull);
            } catch (e) {
                return _.isEqualWith(oldData, newData, (a, b) => (_.isArray(a) || _.isArray(b)) ? handleArrays(a, b) : handleObject(a, b));
            }
            if (!lite)
                return _.isEqualWith(oldData, newData, (a, b) => (_.isArray(a) || _.isArray(b)) ? handleArrays(a, b) : handleObject(a, b));
            else
                return lite;
        }
    }

    private publish(cacheKey: CacheKey, useIndexedDB: boolean = true, pushFromIndex: boolean = true): void {
        cacheKey.executable().pipe(take(1)).subscribe(
            res => {
                this.ngZone.run(() => cacheKey.subject.next(res));
            },
            err => this.ngZone.run(() => cacheKey.subject.error(err)));
    }
