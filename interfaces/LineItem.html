  /**
   * Refreshes a cart based on the given id.
   * @override
   * @param cartId The id of the cart to refresh.
   */
  refreshCart() {
    let attempts = 0;
    return this.userService
      .isLoggedIn()
      .pipe(
        switchMap(loggedIn => {
          const cart$ = this.apiService.get(`/carts/${CartService.getCurrentCartId()}?lookups=${this.cartLookups}&children=${this.cartChildren}&alias=false`, this.type, false);
          const obsv$ = cart$
            .pipe(
              switchMap(cart => {
                if (!isNil(get(cart, 'Id')))
                  return this.getCartLineItems(cart.Id)
                    .pipe(
                      map(lines => {
                        set(cart, 'LineItems', lines);
                        return cart;
                      })
                    );
                else
                  return of(cart);
              })
            );

          return obsv$.pipe(
            map((cart: Cart) => {
              if (loggedIn && !isNil(get(cart, 'Id')))
                CartService.setCurrentCartId(get(cart, 'Id'));
              else if (isNil(cart) && localStorage.getItem(CartService.STORAGE_KEY)) {
                CartService.deleteLocalCart();
                this.refreshCart();
              }

              if (get(cart, 'IsPricePending') === true && this.configService.get('pricingMode') !== 'turbo') {
                if (attempts <= this.configService.get('cartRetryLimit', 3)) {
                  throw new Error('Cart Pending');
                }
              }

              return cart;
            }),
            retryWhen(errors =>
              errors.pipe(
                tap(e => {
                  if (e.message === 'Cart Pending')
                    attempts += 1;
                  else
                    throw e;
                }),
                delay(500)
              )
            )
          );
        }),
        take(1)
      ).subscribe(c => this.publish(c));
  }

  }

  getChildCart(cartId: string = CartService.getCurrentCartId()): Observable<Cart> {
    // API request to get child cart record
    const cartHeader$ = this.apiService.get(`/carts?condition[0]=ParentConfigurationId,Equal,${cartId}&alias=false`, this.type).pipe(map(cartList => first(cartList)));

    // API request to get all line items for child carts related to the given parent cart
    const cartItems$ = this.apiService.get(`/cartitems?condition[0]=Configuration.ParentConfigurationId,Equal,${cartId}&lookups=Product&alias=false`);

    // Combine the lines and the cart header
    return combineLatest([cartHeader$, cartItems$])
      .pipe(map(([cartHeader, cartItems]) => {
        const lineItems = plainToClass(this.metadataService.getTypeByApiName('Apttus_Config2__LineItem__c'), cartItems) as unknown as Array<CartItem>;
        set(cartHeader, 'LineItems', lineItems);
        return cartHeader;
      }));
  }

  private remove(cartId: string | Array<string>): Observable<boolean> {
    const obsv$ =
      cartId instanceof Array
        ? this.apiService.post(
          `/carts/delete`,
          cartId.map(cId => ({ Id: cId }))
        )
        : this.apiService.delete(`/carts/${cartId}`);

    return obsv$.pipe(
      map(deleteList => every(deleteList, l => get(l, 'success') === true)),
      tap(c => this.refreshCart())

  private mergeCartItems(cartItemList: Array<any>) {
    if (!isNil(cartItemList) && !isEmpty(cartItemList)) {
      const keyFn = (r) => r.LineNumber + '.' + r.PrimaryLineNumber + '.' + r.ItemSequence;

      remove(this.state.value.LineItems, (l) => includes(_map(cartItemList, 'LineNumber'), l.LineNumber) && l.LineType === 'Option' && !includes(cartItemList, (i) => keyFn(i) === keyFn(l)));

      this.state.value.LineItems = values(merge(keyBy(this.state.value.LineItems, keyFn), keyBy(cartItemList, keyFn)));

      this.republish(true);

      const configId = get(first(cartItemList), 'ConfigurationId');
      if (!isNil(configId))
        localStorage.setItem(CartService.STORAGE_KEY, configId);
    }
  }

  private onAdd(cartItemList?: Array<any>): void {
    const lineNumbers: Array<number> = uniq(_map(cartItemList, 'LineNumber'));

    this.mergeCartItems(cartItemList);
    if (this.configService.get('skipPricing') === true && !CartService.DeferredPrice)
      this.priceCart(this.configService.get('pricingMode'), lineNumbers);
    else
      this.republish(false);
  }

  addItem(cartRequestList: Array<CartRequest>): Observable<Array<CartItem>> {

  updateItem(cartItemId: string, cartRequestList: Array<CartRequest>, isEmbedded: boolean = false): Observable<Array<CartItem>> {
    // Get child cart record if the bundle is loaded in embedded mode.
    const childCart$ = isEmbedded ? this.getChildCart() : of(null);
    const updateItem$ = childCart$.pipe(
      switchMap((childCart: Cart) => {
        if (isEmbedded && childCart) {
          return this.cloneCart(childCart.Id, CartService.getCurrentCartId())
        } else {
          return this.apiService.put(
            `/carts/${CartService.getCurrentCartId()}/items/${cartItemId}?${this.getQueryParams()}&alias=false`,
            cartRequestList
          );
        }
      }));
    return updateItem$.pipe(
      map(result => {
        if (isEmpty(result.LineItems)) {
          throw 'Failed to update item(s)';
        } else if (isEmbedded) {
          return result.LineItems;
        } else
          return plainToClass(this.metadataService.getTypeByApiName('Apttus_Config2__LineItem__c'), result.LineItems) as unknown as Array<CartItem>;
      }),
      tap(response => this.onAdd(response))
    );
  }

        } else {
          remove(get(this.state, 'value.LineItems', []), o =>
            includes(cartItemIds, o.Id)
          );
        }
      })
    );

    return this.actionQueue.queueAction(removeCartItems$, bind(() => {
      (this.configService.get('skipPricing') && !CartService.DeferredPrice) ? this.priceCart() : this.republish();
      if (CartService.DeferredPrice) this.state.value.set('IsPriced', true);
    }));
  }
  /**
     * priceCart method triggers the server-side pricing engine for the cart
     *
     * ### Example:
```typescript
import { CartService } from '@congacommerce/ecommerce';

export class MyComponent implements OnInit{

    constructor(private cartService: CartService){}

    priceCart(pricingMode): void{
        this.cartApi.priceCart(pricingMode).subscribe(
            () => {...},
            err => {...}
        );
    }
}
```
   * Price active cart and publishing its state to all subscribing components.
   * @param pricingMode Pass the pricing mode you want to run pricing with. It can be turbo OR default pricing.
   */
  priceCart(pricingMode: string = this.configService.get('pricingMode'), lineNumbers?: Array<number>): void {
    let lineParam: string = '';
    if (!isNil(lineNumbers) && isNil(this.state.value.CouponCodes)) {
      lineParam = '&lineNumbers=' + join(compact(lineNumbers), ',');
    }
    this.apiService
      .post(`/carts/${CartService.getCurrentCartId()}/price?mode=${pricingMode}&alias=false${lineParam}`, null, this.type, null, false)
      .pipe(
        take(1)
      )
      .subscribe(priceResponse => {
        if (pricingMode === 'turbo' && typeof priceResponse !== 'boolean') {
          const cart = this.state.value;
          forEach(get(priceResponse, 'LineItems'), (cli) => {
            if (isNil(cli.Id)) {
              const sfCLI = find(cart.LineItems, (li) => (li.ExtendedDescription === cli.ExtendedDescription && li.ExtendedDescription !== null));
              if (sfCLI)
                cli.Id = sfCLI.Id;
            }
            set(cli, 'Product', defaultTo(get(find(cart.LineItems, r => r.Id === cli.Id), 'Product'), get(cli, 'Product')));
            set(cli, 'AttributeValue', defaultTo(get(find(cart.LineItems, r => r.Id === cli.Id), 'AttributeValue'), get(cli, 'AttributeValue')));
            set(cli, 'ProductOption', defaultTo(get(find(cart.LineItems, r => r.Id === cli.Id), 'ProductOption'), get(cli, 'ProductOption')));
          });

          const tbCartItemIds = _map(get(priceResponse, 'LineItems'), item => item.Id);
          forEach(get(cart, 'LineItems'), (cli) => {
            if (includes(tbCartItemIds, cli.Id)) {
              set(cli, 'AdjustmentLineItems', null);
              set(cli, 'IncentiveAdjustmentAmount', null);
              set(cli, 'IncentiveCode', null);
            }
          });

          // New charge line items added due to pricing on the cart.
          const newLines = filter(get(priceResponse, 'LineItems'), lineItem => isNil(lineItem.Id));
          const tpResponse = filter(get(priceResponse, 'LineItems'), item => !isNil(item.Id));
          const mergedItems = compact(values(merge(keyBy(cart.LineItems, 'Id'), keyBy(tpResponse, 'Id'))));

          cart.LineItems = plainToClass(this.metadataService.getTypeByApiName('Apttus_Config2__LineItem__c'), concat(mergedItems, newLines), { ignoreDecorators: true }) as any;
          set(cart, 'SummaryGroups', get(priceResponse, 'SummaryGroups'));

          if (!cart.Id)
            cart.Id = CartService.getCurrentCartId();
          cart.IsPricePending = false;
          if (CartService.DeferredPrice) cart.set('IsPriced', false);
          this.publish(cart);
        } else {
          this.refreshCart();
        }
      });
  }

  getCartWithId(cartId: string): Observable<Cart> {
    return this.get([cartId]).pipe(map(first));
  }

  abandonCart(cartId: string): Observable<boolean> {
    return this.platformService.apiService.post(`/carts/${cartId}/abandon`);
  }

  /**
 * To render the iFrame, a business object Id must be determined by the client.
 * The Business Object Id can be retrieved from a child cart associated with the primary cart.
 * To retrieve this child cart, this method will be invoked.
 * @param lineNo lineitem line number which needs to be edited.
 */
  childCart(): Observable<Cart> {
    return this.getMyCart()
      .pipe(
        take(1),
        switchMap(data => (isNil(get(data, 'Id'))) ? this.createNewCart() : of(data)),
        mergeMap((cart: Cart) => this.apiService.get(`/carts/${cart.Id}/children`)));
  }

  /**
   * This method gives right state of the cart pricing to client.
   * Returns boolean response,
   * TRUE: Cart is price pending and recent attempt to price the cart has failed.
   * FALSE: Cart is succesfully priced and recent attempt to price the cart is succesfull.
   */
  getCartPriceStatus(): Observable<boolean> {
    return this.getMyCart()
      .pipe(
        map(c => {
          return c.hasErrors && filter(c.errors, er => er.message === 'Pricing Failed').length > 0
            && (c.IsPricePending || c.LineItems.filter(i => i.PricingStatus !== 'Complete').length > 0);
        }));
  }

  /**
   * This method clones a list of primary cart lines.
   * The cloned items get added to the active cart.
   * @param lineItems Array of cart items to be cloned on the active cart.
   * @returns observable of cloned cart items.
   */
  clonePrimaryLines(lineItems: Array<CartItem>): Observable<Array<CartItem>> {
    const lineNumbers = _map(lineItems, item => item.LineNumber);
    return this.getMyCart().pipe(
      take(1),
      switchMap(cart => this.apiService.post(`/carts/${get(cart, 'Id')}/items/clone`, lineNumbers)),
      map(res => {
        const resp = plainToClass(this.metadataService.getTypeByApiName('Apttus_Config2__LineItem__c'), res, { ignoreDecorators: true }) as unknown as Array<CartItem>;
        return resp;
      }),
      tap(lineItems => this.onAdd(lineItems))
    )
  }
