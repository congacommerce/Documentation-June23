    aqlResultToMetadata(result: Object): Metadata {

        metadataObj.Description = _.get(result, 'Description');

            objectAttributes.length = metadataFields[key].Length;
            objectAttributes.type = metadataFields[key].Type;
            objectAttributes.name = metadataFields[key].Name;
            const type = metadataFields[key].Type;
            switch (type) {
                case 'SelectOption': {

                        let picklist: Array<Picklist> = [];
                        objectAttributes.type = 'picklist';
                        let selectOptions = metadataFields[key].SelectOptionSet.SelectOptions;

                }

    private getAQLCriteria(conditionList?: Array<ACondition>, expressionOperator: 'AND' | 'OR' = 'AND', entity?: AObject): Object {
        let conditions = [
            {
                FieldName: 'Id',
                EntityName: entity.getApiName(),
                FilterOperator: 'NotNull',
                Value: null,
                IsValueFieldName: false
            }
        ];
        conditions = conditions.concat(this.getAQLConditions(conditionList));

    private getAQLJoins(instance: AObject, joinList: Array<AJoin>): Array<AQLJoin> {
        const _joins = _.unionBy(joinList, 'type').filter(join => join != null);
        const aqlJoinList: Array<AQLJoin> = [];
        // right
        if (_.get(_joins, 'length', 0) > 0) {
            _joins.forEach(join => {
                aqlJoinList.push({
                    JoinType: 'RIGHT',
                    EntityAlias: new join.type().getApiName(),
                    FromEntity: instance.getApiName(),
                    ToEntity: new join.type().getApiName(),
                    FromAttribute: instance.getApiName(join.fromAttribute),
                    ToAttribute: new join.type().getApiName(join.toAttribute),
                    JoinCriteria: {
                        ExpressionOperator: join.expressionOperator,
                        Conditions: this.getAQLConditions(join.joinCriteria, null, new join.type().getApiName()),
                        Filters: this.getAQLFilters(join.filterList, new join.type().getApiName())
                    }
                });
            });
        }
        return aqlJoinList;
    }

    private aqlToSOQLCondition(aObject: AObject, conditionList?: Array<ACondition>, filterList?: Array<AFilter>, expressionOperator: 'AND' | 'OR' = 'AND', joinList?: Array<AJoin>, sortOrder?: Array<ASort>, pageInfo?: APageInfo, groupBy?: Array<string>) {
        let _joinList = [];
        _.values(_.groupBy(joinList, 'type')).forEach(data => {
            _joinList = _joinList.concat(mergeRecords(<Array<any>>data));
        });
        joinList = _joinList;
        let queryString = 'WHERE ID <> NULL ';
        // criteria
        if (_.get(conditionList, 'length', 0) > 0 || _.get(joinList, 'length', 0) > 0 || _.get(filterList, 'length', 0) > 0)
            queryString += 'AND ';

        if (_.get(conditionList, 'length', 0) > 0) {
            queryString += `(${this.conditionListToSOQL(conditionList, aObject, expressionOperator)})`;
        }

        // filter list
        if (_.get(filterList, 'length', 0) > 0) {
            if (_.get(conditionList, 'length', 0) > 0)
                queryString += ` ${expressionOperator} `;
            const filterString = this.filterListToSOQL(filterList, expressionOperator);
            queryString += filterString;
        }

        if (_.get(joinList, 'length', 0) > 0) {
            if (_.get(conditionList, 'length', 0) > 0 || (_.get(filterList, 'length', 0) > 0))
                queryString += ` ${expressionOperator} `;

            // Merge similar joins
            joinList.forEach((x, xIndex) => {
                joinList.forEach((y, yIndex) => {
                    if (_.get(y, 'fromAttribute') === _.get(x, 'fromAttribute') && _.get(y, 'toAttribute') === _.get(x, 'toAttribute') && _.get(y, 'type') === _.get(x, 'type') && yIndex !== xIndex) {
                        x.conditions = (x.conditions) ? _.get(x, 'conditions', []).concat(y.conditions) : y.conditions;
                        x.filters = (x.filters) ? _.get(x, 'filters', []).concat(y.filters) : y.filters;
                        joinList.splice(yIndex, 1);
                    }
                });
            });

            joinList.forEach((join, index) => {
                queryString += this.joinToSOQL(join, aObject);
                if (index < (joinList.length - 1))
                    queryString += ` ${expressionOperator} `;
            });
        }

        // Group By
        if(!_.isEmpty(groupBy))
            queryString += ` ${this.groupByListToSOQL(aObject, groupBy)}`;

        // Order By
        if (_.get(sortOrder, 'length', 0) > 0)
            queryString += `${this.sortOrderListToSOQL(sortOrder, !_.isEmpty(groupBy))}`;

        queryString += `${this.aPageInfoToSOQL(pageInfo)}`;

        return queryString;
    }

            (sort, index) => {

                if (fieldName) {
                    if(aggregate){

    private conditionListToSOQL(conditionList: Array<ACondition>, instance: ClassType<AObject> | AObject, expressionOperator: 'AND' | 'OR' = 'AND'): string {
        let queryString = '';
        const obj = (instance instanceof AObject) ? instance : new instance();

        if (conditionList && conditionList.length > 0) {
            conditionList.forEach((condition, index) => {
                if (condition) {
                    let value = condition.val;
                    const fieldName = obj.getApiName(condition.field);
                    const stringOperators = [Operator.CONTAINS, Operator.DOES_NOT_CONTAIN, Operator.BEGINS_WITH, Operator.DOES_NOT_BEGIN_WITH, Operator.ENDS_WITH, Operator.DOES_NOT_END_WITH];
                    if (value instanceof Array) {
                        value = AParser.arrayToCsv(value);
                    }
                    else if (value instanceof SimpleDate) {
                        value = value.getSimpleDate();
                    }
                    else if (value instanceof Date) {
                        value = value.toISOString();
                    }
                    else if (typeof value === 'string' && !AParser.isLiteral(value)) {
                        value = `${_.escape(value)}`;
                        if (!_.includes(stringOperators, condition.filterOperator))
                            value = `'${value}'`;
                    }
                    const operatorMap = {
                        'Equal': `${fieldName} = ${value}`,
                        'NotEqual': `${fieldName} <> ${value}`,
                        'GreaterThan': `${fieldName} > ${value}`,
                        'LessThan': `${fieldName} < ${value}`,
                        'GreaterEqual': `${fieldName} >= ${value}`,
                        'LessEqual': `${fieldName} <= ${value}`,
                        'Like': `${fieldName} LIKE ${value}`,
                        'NotLike': `(NOT ${fieldName} LIKE ${value})`,
                        'Contains': `${fieldName} LIKE '%${value}%'`,
                        'DoesNotContain': `(NOT ${fieldName} LIKE '%${value}%')`,
                        'BeginsWith': `${fieldName} LIKE '${value}%'`,
                        'DoesNotBeginWith': `(NOT ${fieldName} LIKE '${value}%')`,
                        'EndsWith': `${fieldName} LIKE '%${value}'`,
                        'DoesNotEndWith': `(NOT ${fieldName} LIKE '%${value}')`,
                        'In': `${fieldName} IN (${value})`,
                        'NotIn': `${fieldName} NOT IN (${value})`,
                        'Null': `${fieldName} = NULL`,
                        'NotNull': `${fieldName} <> NULL`,
                        'On': `${fieldName} = ${value}`,
                        'LastXDays': `${fieldName} = LAST_N_DAYS:${value}`,
                        'NextXDays': `${fieldName} = NEXT_N_DAYS:${value}`,
                        'Today': `${fieldName} = TODAY`,
                        'Includes': `${fieldName} INCLUDES (${value})`
                    };

                    const query = operatorMap[condition.filterOperator];
                    queryString += query;
                    if (index < (conditionList.length - 1))
                        queryString += ` ${expressionOperator} `;
                }
            });
        }
        return queryString;
    }

    private filterListToSOQL(filterList: Array<AFilter>, expressionOperator: 'AND' | 'OR' = 'AND') {
        let queryString = '';
        filterList = _.compact(filterList);
        _.forEach(filterList, (filter, index) => {
            if (_.get(filter, 'conditions.length', 0) > 0 || _.get(filter, 'filters.length', 0) > 0) {
                if (_.get(filter, 'conditions.length', 0) > 0) {
                    queryString += `(${this.conditionListToSOQL(filter.conditions, filter.type, filter.expressionOperator)})`;
                }
                if (_.get(filter, 'filters.length', 0) > 0) {
                    if (_.get(filter, 'conditions.length', 0) > 0)
                        queryString += ` ${expressionOperator} `;
                    queryString += `(${this.filterListToSOQL(filter.filters, filter.expressionOperator)})`;
                }

                if (index < (filterList.length - 1)) {
                    queryString += ` ${expressionOperator} `;
                }
            }
        });
        return queryString;
    }

    private getAggregateFields(type: AObject, prefix: string, adtlFields?: Array<string>): Array<string> {
        let fieldArray = [];
        if(!prefix && _.isNil(_.find(adtlFields, f => _.lowerCase(f) === 'id')))
            fieldArray.push('COUNT(Id) total_records');

        const aggregateFields = type.getFieldMetadata(m => !_.isNil(_.get(m, 'aggregate')));
        _.forEach(_.keys(aggregateFields), (field) => {
            _.forEach(_.get(aggregateFields, `${field}.aggregate`), aggregateType => {
                fieldArray.push(`${aggregateType}(${type.getApiName(field)}) ${aggregateType}_${field}`);
            });
        });

        if (adtlFields && adtlFields.length > 0){
            fieldArray = _.concat(fieldArray, _.map(adtlFields, f => type.getApiName(f)));
        }
        return fieldArray;
    }

    @memoizeAll((...args) => JSON.stringify([args[0].generatedId(), args.slice(1)]))
    private getFields(instance: AObject, prefix: string, depth: number = 0, ignoreJoins: boolean = false, customFields: Array<string> = [], childFilters: Array<AChildFilter> = [], groupBy?: Array<string>, multiCurrencyEnabled: boolean = false, flat: boolean = false): Array<string> {
        const sfdcDefaultFields = ['CreatedDate', 'LastModifiedDate', 'LastModifiedById', 'CreatedById'];
        const aicDefaultFields = ['CreatedOn', 'ModifiedOn', 'ModifiedById', 'CreatedById'];

        const defaultFields = (this.configurationService.get('type') === 'Salesforce') ? sfdcDefaultFields : aicDefaultFields;
        const currencyFields = ['DefaultCurrencyIsoCode', 'CurrencyIsoCode', 'CountryCode', 'StateCode'];
        const keys = this.getFieldNamesForInstance(instance);
        let fieldArray: Array<string> = new Array<string>();

        if(!_.isNil(groupBy)){
            fieldArray = _.map(groupBy, f =>
                `${instance.getApiName(f, 'Salesforce')} ${_.replace(instance.getApiName(f, 'Salesforce'), '.', '___')}`
            );
        }else{
            const metadata = instance.getMetadata();

            for (const key of keys) {
                const _key = instance.getApiName(key);

                if (_key === undefined || _key === 'undefined')
                    continue;

                if (!multiCurrencyEnabled) {
                    let process = true;
                    for (const cField of currencyFields) {
                        if (key.indexOf(cField) >= 0) {
                            process = false;
                        }
                    }
                    if (!process)
                        continue;
                }
                if (metadata && _.get(metadata, 'defaultFields') === false && defaultFields.indexOf(key) >= 0)
                    continue;

                if ((<string>key).charAt(0) === '_')
                    continue;

                const fieldType = instance.getReferenceType(key);
                const field = instance.instanceOf(key);

                if (flat === true && (fieldType === 'Child' || fieldType === 'Lookup'))
                    continue;

                if (fieldType === 'Child' && !prefix && !ignoreJoins && field) {
                    fieldArray.push(this.getChildFields(field, _key, _.filter(childFilters, filter => key === filter.field)));
                } else if (fieldType === 'Lookup' && field) {
                    const newPrefix = (prefix) ? prefix + '.' + <string>_key : <string>_key;
                    if (depth < this.configurationService.get('queryDepth')) {
                        fieldArray = fieldArray.concat(this.getFields(field, newPrefix, depth + 1, false));
                    } else {
                        continue;
                    }
                } else if (!fieldType) {
                    if (prefix)
                        fieldArray.push(prefix + '.' + <string>_key);
                    else
                        fieldArray.push(<string>_key);
                }
            }
            if (depth === 0 && !ignoreJoins && !prefix && customFields && customFields.length > 0)
                fieldArray = fieldArray.concat(customFields);
        }
        return _.uniqBy(fieldArray, _.lowerCase);
    }

    private getChildFields(type: AObject, name: string, childFilters: Array<AChildFilter>): string {
        let str = `(SELECT `;
        let fieldArray = this.getFields(type, null, 0, true);
        const childFilter = _.first(childFilters);

        fieldArray = fieldArray.filter(e => (e.charAt(0) !== '_'));

        str += fieldArray.join(',');
        str += ` FROM ${name}`;
        if (_.get(childFilters, 'length', 0) > 0) {
            const condition = this.filterListToSOQL(_.flatten(_.map(childFilters, 'filterList')));
            if (condition)
                str += ` WHERE ${condition}`;
        }

        if(_.get(type, '_order') || _.get(childFilter, 'sortField'))
            str += ` ORDER BY ${_.defaultTo(_.get(type, '_order'), type.getApiName(_.get(childFilter, 'sortField')))} ${_.get(childFilter, 'order', 'ASC')}`;
        str += `${this.aPageInfoToSOQL(_.get(childFilter, 'page'))}`;

        str += `)`;

        return str;
    }
}

/**
 * Simple object check.
 * @param item
 */
export function isObject(item) {
    return (item && typeof item === 'object' && !Array.isArray(item));
}

/**
 * Deep merge two objects.
 * @param target
 * @param ...sources
 */
export function mergeDeep(target, ...sources) {
    if (!sources.length) return target;
    const source = sources.shift();

    if (isObject(target) && isObject(source)) {
        for (const key in source) {
            if (isObject(source[key])) {
                if (!target[key]) Object.assign(target, { [key]: {} });
                mergeDeep(target[key], source[key]);
            } else {
                Object.assign(target, { [key]: source[key] });
            }
        }
    }

    return mergeDeep(target, ...sources);
}

export function mergeRecords(records: Array<any>): any{
    const joined = [];
    if (_.get(records, 'length', 0) > 0) {
        let y;
        (<Array<any>>records).forEach(record => {
            if (!y)
                y = record;
            else
                mergeDeep(y, record);
        });
        joined.push(y);
    }
    return joined;
}
