import { of, Observable, ReplaySubject, from, throwError, forkJoin } from 'rxjs';
import { catchError, take, flatMap, map, switchMap, mergeMap, tap } from 'rxjs/operators';
import { PlatformConstants } from '../classes/configuration.model';
import { ConfigurationService } from './configuration.service';
import * as jsforce from 'jsforce/build/jsforce.min.js';
import { JSForceService } from './jsforce.service';
import { CacheService } from './cache.service';
import * as _ from 'lodash';
import { Composite } from '../interfaces/composite.interface';
import { AObject } from '../classes/index';
import { MetadataService } from './metadata.service';
import { HttpClient } from '@angular/common/http';
import { ClassType } from 'class-transformer/ClassTransformer';
import { EventEmitter, Injectable } from '@angular/core';
import { QueryOptions } from '../interfaces';
import { ApiService } from './api.service';

declare var window: any;

let _windowConfig = window.sv || {};

@Injectable({
    providedIn: 'root'
})
export class PlatformService{
    private static STOP_EXCEPTIONS = ['INVALID_SESSION_ID', 'API_CURRENTLY_DISABLED', 'ERROR_HTTP_403', 'ERROR_HTTP_504', 'ERROR_HTTP_400'];

    loginChange: ReplaySubject<any> = new ReplaySubject<any>();
    loggingIn: boolean = false;
    stopOnException: boolean = false;
    connection: any;
    anonymousConnection: any;
    onLogout: EventEmitter<boolean> = new EventEmitter<boolean>();

    private recordMap = {};

    private static lengthInBytes(str) {
        if (!_.isNil(str)) {
            const m = encodeURIComponent(str).match(/%[89ABab]/g);
            return str.length + (m ? m.length : 0);
        } else
            return 0;
    }

    constructor(protected httpClient: HttpClient, protected _cache: CacheService, protected metadataService: MetadataService, protected config: ConfigurationService, public apiService: ApiService) {
        let proxyLogin = this.config.get('proxy');
        if (proxyLogin && !proxyLogin.endsWith('/'))
            proxyLogin += '/';
        else if (!proxyLogin)
            proxyLogin = '';

        const organizationId = (this.config.get('organizationId') ? this.config.get('organizationId') : _windowConfig.organizationId);
        JSForceService.fixSoapLogin(jsforce, organizationId, this.config.proxyEndpoint());
        JSForceService.fixIdentity(jsforce, this.config.proxyEndpoint());
        JSForceService.addSoapCallout(jsforce, this.config.get('production'));
        // JSForceService.fixPrompt(jsforce);
        JSForceService.fixLogout(jsforce);


        // Generate search params for url fragment
        const fragmentParams = new URLSearchParams(window.location.hash.substring(1));

        // If access token is in visualforce page
        if (_windowConfig && _windowConfig.accessToken && _windowConfig.accessToken !== 'NULL_SESSION_ID') {
            localStorage.setItem(PlatformConstants.ACCESS_TOKEN, _windowConfig.accessToken);
        }
        // If access token is in url fragment
        else if (!_.isNil(fragmentParams.get(PlatformConstants.ACCESS_TOKEN))) {
            localStorage.setItem(PlatformConstants.ACCESS_TOKEN, fragmentParams.get(PlatformConstants.ACCESS_TOKEN));
        }
        this.apiService.onRefresh.subscribe(token => this.setAccessToken(token));
        this.resetConnection();
    }

    setPassword(password: string) {
        return this.userId().pipe(flatMap(userId => {
            return this.connection.soap.setPassword(userId, password);
        }));
    }

    getConnection(): any {
        return this.connection;
    }

    login(username: string, password: string): Observable<any> {
        if (!username || !password) {
            let credentialKey = localStorage.getItem(PlatformConstants.KEY);
            if (credentialKey) {
                let storedCredentials = JSON.parse(atob(credentialKey));
                if (storedCredentials && storedCredentials.username && storedCredentials.password)
                    return this.doLogin(storedCredentials.username, storedCredentials.password);
                else
                    return throwError(new Error('Invalid Login'));
            } else {
                return throwError(new Error('Invalid Login'));
            }
        } else {
            return this.doLogin(username, password);
        }
    }

    loginOauth(): void {
        jsforce.browser.init({
            clientId: this.config.get('connectedApp.clientId'),
            redirectUri: this.config.get('connectedApp.redirectUri', this.config.resourceLocation()),
            loginUrl: this.config.endpoint()
        });
        jsforce.browser.login();
    }

    logout(): Observable<boolean> {
        return this.apiService.revokeToken()
            .pipe(
                catchError(() => of(null)),
                tap(success => {
                    localStorage.clear();
                    if (_windowConfig)
                        delete _windowConfig.accessToken;
                    delete this.connection;
                    this.resetConnection();
                    this.onLogout.emit(true);
                    this._cache.refresh('User');
                })
            );
    }

    setAccessToken(accessToken: string): void {
        if (!_.isNil(accessToken)) {
            this.connection.accessToken = accessToken;
            localStorage.setItem(PlatformConstants.ACCESS_TOKEN, accessToken);
        } else {
            delete this.connection.accessToken;
            localStorage.removeItem(PlatformConstants.ACCESS_TOKEN);
        }
    }

    setUserInfo(userInfo: any) {
        this.connection.instanceUrl = userInfo.instance_url;
        localStorage.setItem(PlatformConstants.USER_INFO, JSON.stringify(userInfo));
    }

    isLoggedIn() {
        return (this.connection && this.connection.accessToken) ? true : false;
    }

    isGuest(): Observable<boolean> {
        return of(!this.connection.accessToken);
    }

    create(objectList: Array<any>, objectType: ClassType<AObject>, updateCache?: boolean): Observable<Array<string>> {
        const apiName = new objectType().getApiName();
        return this.apiService.post(apiName, objectList, objectType)
            .pipe(
                map((res: Array<any>) => res.map(a => _.get(a, 'Id', _.get(a, 'id'))))
            );
    }

    remove(objectList: Array<any>, objectType: new (t?) => AObject, updateCache?: boolean): Observable<Array<string>> {
        const apiName = new objectType().getApiName();
        return this.apiService.post(`${apiName}/delete`, objectList).pipe(
            map((res: Array<any>) => res.map(a => _.get(a, 'Id', _.get(a, 'id'))))
        );
    }

    update(objectList: Array<any>, objectType: ClassType<AObject>, updateCache?: boolean): Observable<Array<string>> {
        const apiName = new objectType().getApiName();
        return this.apiService.put(apiName, objectList, objectType).pipe(
            map((res: Array<any>) => res.map(a => _.get(a, 'Id', _.get(a, 'id'))))
        );
    }

    upsert(objectList: Array<any>, objectType: new (t?) => AObject, updateCache?: boolean): Observable<Array<string>> {
        const apiName = new objectType().getApiName();
        return this.apiService.put(apiName, objectList, objectType).pipe(
            map((res: Array<any>) => res.map(a => _.get(a, 'Id', _.get(a, 'id'))))
        );
    }

    composite(dataList: Array<Composite>) {
        return this.post('composite', { 'q': dataList.filter(d => d.data && d.method && d.sobjectType) }).pipe(take(1));
    }

    callout(url: string, method: 'GET' | 'POST' | 'PATCH' | 'PUT' | 'DELETE' = 'GET', body?: object): Observable<any> {
        return from(this.getConnection().request({
            method: method,
            url: url,
            body: body
        }));
    }

    post(method, data, classOverride: string = 'DataHandler', anonymous: boolean = false, count: number = 0, namespace: string = this.config.get('packageNamespace')): Observable<any> {
        const body = {
            class: classOverride,
            method: method,
            data: data,
            encryptedResponse: this.config.get('encryptResponse'),
            namespace: namespace
        };
        const start = new Date().getTime();
        const location = !_.isNil(this.config.get('packageNamespace')) ? `/${this.config.get('packageNamespace')}/dc/` : '/dc/';
        const connection = (anonymous) ? this.anonymousConnection : this.connection;
        return this._cache.encrypt(body)
            .pipe(
                mergeMap(payload => from(connection.apex.post(location + start, { d: payload }))
                    .pipe(
                        mergeMap((res: string) => (this.config.get('encryptResponse')) ? this._cache.decrypt(res) : of(JSON.parse(res))),
                        map((res: any) => {
                            const end = new Date().getTime();
                            if (res && res.success) {
                                this.logRequest(payload, body, (end - start), res);
                                try {
                                    return JSON.parse(res.data);
                                } catch (e) {
                                    return (res.data) ? res.data : res;
                                }
                            } else {
                                throw (res && res.data) ? res.data : null;
                            }
                        })
                        , catchError(e => {
                            const end = new Date().getTime();
                            this.logRequest(payload, body, (end - start));
                            if (PlatformService.STOP_EXCEPTIONS.indexOf(e.name) >= 0 && count <= 2 && !_.isNil(this.connection.accessToken)) {
                                count += 1;
                                return this.logout().pipe(switchMap(() => this.post(method, data, classOverride, false, count)));
                            }
                            else
                                return throwError(e);
                        })
                    )
                ),
                take(1)
            );
    }

    logRequest(payload, body, requestTime, response?) {
        if (this.config.get('production') !== true) {
            if ((this.config.get('enableQueryLogs') === true && body) || (this.config.get('enablePerformanceLogs') === true && requestTime)) {
                // tslint:disable-next-line:no-console
                console.debug('-------------------------------');
                if (this.config.get('enableQueryLogs') === true) {
                    // tslint:disable-next-line:no-console
                    console.debug(body);

                    if (_.get(body, 'data.q'))
                        // tslint:disable-next-line:no-console
                        console.debug(`Objects ${_.uniq(_.map(_.values(_.get(body, 'data.q')), a => _.get(a, 'type')))}`);

                    if (!_.isNil(response)) {
                        let responseData = response.data;
                        try {
                            responseData = JSON.parse(_.get(response, 'data', '{}'));
                        } catch (e) {
                            // suppress
                        }
                        finally {
                            _.forEach(responseData, recordSet => _.forEach(recordSet, r => {
                                if (!_.isNil(_.get(r, 'Id')))
                                    _.set(this.recordMap, _.get(r, 'Id'), _.get(this.recordMap, _.get(r, 'Id'), 0) + 1);
                            }));
                            const max = _.findKey(this.recordMap, o => _.max(_.values(this.recordMap)) === o);
                            // tslint:disable-next-line:no-console
                            console.debug(`Record with id ${max} has been queried for ${_.get(this.recordMap, max)} times`);
                        }
                    }
                }
                if (this.config.get('enablePerformanceLogs')) {
                    // tslint:disable-next-line:no-console
                    console.debug(`Type: ${body.method}    Input: ${PlatformService.lengthInBytes(payload)} bytes    Output: ${PlatformService.lengthInBytes(JSON.stringify(response))} bytes    Time: ${requestTime} milliseconds`);
                }
            }
        }
    }


    query(options: QueryOptions, type: ClassType<AObject>, parse: boolean = false){
        const apiName = new type().getApiName();
        if(parse)
            return this.apiService.post(`/${apiName}/query`, options, type);
        else
            return this.apiService.post(`/${apiName}/query`, options);
    }

    checkChildren(data): Observable<Array<any>> {
        let obsvArray = [];
        for (let record of data.records) {
            for (const property in record) {
                if (record.hasOwnProperty(property) && record[property].done === false)
                    obsvArray.push(this.queryMore(record[property].nextRecordsUrl, record[property], record, property));
            }
        }
        if (obsvArray && obsvArray.length > 0)
            return forkJoin(obsvArray);
        else
            return of(data);
    }

    queryMore(locator: string, data, parent, parentProperty): Observable<Array<any>> {
        return from(this.connection.queryMore(locator)).pipe(flatMap((res: any) => {
            res.records = res.records.concat(data.records);
            if (parent && parent[parentProperty])
                parent[parentProperty] = res;
            if (res.done === false)
                return this.queryMore(res.nextRecordsUrl, res, parent, parentProperty);
            else
                return of(res);
        }));
    }

    describe(sobject: string, fieldName: string, picklistOnly: boolean): Observable<any> {
        return this.apiService.get(`${sobject}/describe`).pipe(map(_.first));
    }

    userId(): Observable<string> {
        if (localStorage.getItem(PlatformConstants.USER_INFO)) {
            const userInfo = JSON.parse(localStorage.getItem(PlatformConstants.USER_INFO));
            const id = _.get(userInfo, 'id', '');
            return of(id.substring(id.lastIndexOf('/') + 1));
        } else {
            // return this.post('identity', {});
            return this.getUserData()
                .pipe(
                    map(data => data.userId)
                );
        }
    }

    getUserData(username?: string, password?: string) {
        return this.post('getAccessToken', {
            storefrontName: this.config.get('storefront'),
            key: username,
            value: password
        })
            .pipe(
                tap(data => {
                    if (!_.isNil(data))
                        this.setAccessToken(data.accessToken);
                })
            );
    }

    organizationId() {
        return this.post('organization', {});
    }

    private resetConnection() {

        let instanceUrl = 'https://' + window.location.hostname + window.location.pathname;

        if (_.get(_windowConfig, 'baseURL', '').trim().length > 0) {
            instanceUrl = 'https://' + _windowConfig.baseURL;
        } else if (_.get(_windowConfig, 'salesforceEndpoint', '').trim().length > 0) {
            instanceUrl = 'https://' + _windowConfig.salesforceEndpoint + window.location.pathname;
        } else {
            instanceUrl = this.config.endpoint();
        }

        const args = {
            loginUrl: instanceUrl,
            instanceUrl: instanceUrl,
            serverUrl: instanceUrl
        };
        this.connection = new jsforce.Connection(args);
        this.anonymousConnection = new jsforce.Connection(args);

        // Set the access Token
        this.connection.accessToken = localStorage.getItem(PlatformConstants.ACCESS_TOKEN);
    }

    private doLogin(username: string, password: string): Observable<any> {
        return this.apiService.refreshToken(username, password);
    }
